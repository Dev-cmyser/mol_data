}
export function parallel<V>(v: Wrapped<V>): V {
  const keys = Object.keys(v)
  const result = {} as V
  const promises: PromiseLike<any>[] = []
  for (let key of keys) {
    try {
      result[key as keyof V] = v[key]()
    } catch (error) {
      if (isPromise(error)) promises.push(error)
